함수(function)
    - 자바스크립트 코드 저장
    - function 함수이름 { }

일반함수는 함수 호출 시 호이스팅 기술을 지원, 익명함수는 선언 참조 방식은 지원하지 않음
호이스팅을 적용하면 함수 정의문보다 호출문이 먼저 나와도 함수 정의문을 끌어 올려 함수르 호출

데이터를 반환하고 강제종료하는 return 문


재귀함수
    - 함수 안에 본인함수가 호출

함수 스코프
    - 전역 변수, 지역 변수
        var 변수명;         // 전역변수
        function 함수명(){
            var 변수명;     // 지역 변수 : 함수 안에서만 사용
        }
    - 전역 함수, 지역 함수
        function 함수명(){          // 전역 함수
            코드;
        }

        function 함수명(){
            function 함수명(){      // 지역 함수
                코드;
            }
        }

    - 즉시 실행 함수
        (function(){
            코드;
        }());
        // p148 참고

객체 생성자 함수
    - new 키워드 사용
        function 함수명(매개변수1,매개변수2,...,매개변수n){ // 객체 생성자 함수
            this.속성명 = 값;
            this.함수명 = function(){
                코드;
            }
        }
        var 참조 변수(인스턴스 네임) = new 함수명();        // 객체 생성
        var 참조 변수 = { 속성 : 새 값, 함수명 = function(){...}}
    - 메모리 절약을 위한 프로토타입 사용
        function 함수명(매개변수1,매개변수2,...,매개변수n){
            this.속성명 = 새 값;
        }
        함수명.prototype.함수명 = function(){
            코드;
        }
        var 참조 변수(인스턴스 네임) = new 함수명();
